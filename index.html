// Tampermonkey HN colorizer example:
(function() {
    'use strict';
    const colors = ['#ff6600', '#2196f3', '#4caf50', '#9c27b0', '#f44336'];
    const indentations = document.querySelectorAll('td.ind img');

    indentations.forEach(img => {
        const level = parseInt(img.width) / 40;
        if (level > 0) {
            const commentRow = img.closest('tr');
            const commentText = commentRow.querySelector('.comment').style;
            if (commentText) {
                commentText.borderLeft = `3px solid ${colors[level % colors.length]}`;
                commentText.paddingLeft = '15px';
                commentText.marginLeft = '5px';
            }
        }
    });
})();

#!/bin/bash
# A simple script to check disk usage and warn if high
THRESHOLD=90
USAGE=$(df / | tail -1 | awk '{print $5}' | sed 's/%//')

if [ "$USAGE" -gt "$THRESHOLD" ]; then
    echo "Warning: Disk usage is at ${USAGE}%!"
else
    echo "Disk usage is normal: ${USAGE}%"
fi
